```js
// 获取权限
      try {
        navigator.clipboard.readText()
          .then(text => {
            this.isPasetDisabled = !text
          })
          .catch(() => {
            this.isPasetDisabled = true
            console.log('无粘贴权限')
          })
      } catch (e) {
        this.isPasetDisabled = true
        console.log('需要升级浏览器或更换浏览器', e)
      }
```
